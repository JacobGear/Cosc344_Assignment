1. grant select on emp to jgear1;
2. grant update(dname) on dept to jgear1;
3. revoke all on emp from jgear1;
   revoke all on dept from jgear1;
4. create role seeTables;
   grant select emp to seeTables;
   grant select dept to seeTables;
   grant seeTables to jgear1;
5. create view empDepartment as select emp.fname, emp.lname, dept.dnumber from emp inner join dept on emp.dno=dept.dnumber;
   grant select on empDepartment to jgear1;
6. create view seeSalaries as select fname, lname, salary from emp where salary < 50001;
   grant select on seeSalaries to jgear1;
